class MHS5200A(object):
    def __init__(self, port="/dev/ttyUSB2"):
        cfg=dict()
        cfg["port"]=port
        cfg["baudrate"]=57600
        cfg["xonxoff"]=False        
        cfg["timeout"]=0.5
        cfg["rtscts"]=True
        cfg["dsrdtr"]=False
        self.cfg = cfg
        self.serial = serial.Serial(**cfg)
        
    def send(self, msg="", return_line = False):
        self.serial.write((msg+"\r\n").encode())
        if return_line:
            return self.serial.readline()
        
    def get(self, )
    
        
    def init(self):
        self.send(":")
        self.send(":r1c")
        self.send(":r2c")
